Homebrew MySQL

#fix ownership on /usr/local/var/mysql
	chown -RL mysql:mysql /usr/local/var/mysql
#set proper UserName on 
	defaults write /usr/local/Cellar/mysql/5.5.29/homebrew.mxcl.mysql.plist UserName mysql
